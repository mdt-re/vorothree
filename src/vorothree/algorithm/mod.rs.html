8/// Trait defining a spatial acceleration structure.
9/// This allows swapping between Grid, Octree, or Linear Octree algorithms.
10pub trait SpatialAlgorithm<const D: usize>: Send + Sync {
11    /// Rebuild the index with new generators.
12    fn set_generators(&mut self, generators: &[f64], bounds: &BoundingBox<D>);
13
14    /// Update the position of a single generator.
15    fn update_generator(&mut self, index: usize, old_pos: &[f64; D], new_pos: &[f64; D], bounds: &BoundingBox<D>);
16
17    /// Visit potential neighbors for a given generator.
18    ///
19    /// # Arguments
20    /// * `generators` - The full list of generators (needed to retrieve neighbor positions).
21    /// * `index` - The index of the generator we are processing.
22    /// * `pos` - The position of the generator (array of size D).
23    /// * `max_dist_sq` - A mutable reference to the current maximum search radius squared.
24    ///                   The visitor can update this if the cell shrinks.
25    /// * `visitor` - A closure called for each candidate neighbor. It receives the neighbor's index
26    ///               and its position.
27    fn visit_neighbors<F>(&self, generators: &[f64], index: usize, pos: [f64; D], max_dist_sq: &mut f64, visitor: F)
28    where
29        F: FnMut(usize, [f64; D], f64) -> f64;
30}
