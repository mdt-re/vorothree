4/// A spatial index based on a uniform grid for 2D space.
5///
6/// This structure divides the 2D space into a fixed number of bins.
7/// It allows for O(1) insertion and update operations.
8pub struct AlgorithmGrid2D {
9    /// Number of bins along the X axis.
10    pub grid_res_x: usize,
11    /// Number of bins along the Y axis.
12    pub grid_res_y: usize,
13    /// Scale factor for X coordinate to grid index.
14    pub grid_scale_x: f64,
15    /// Scale factor for Y coordinate to grid index.
16    pub grid_scale_y: f64,
17    /// Maximum valid index for X.
18    pub grid_limit_x: f64,
19    /// Maximum valid index for Y.
20    pub grid_limit_y: f64,
21    /// Minimum X coordinate of the grid bounds.
22    pub min_x: f64,
23    /// Minimum Y coordinate of the grid bounds.
24    pub min_y: f64,
25    /// The grid bins, each containing a list of generator indices.
26    pub grid_bins: Vec<Vec<usize>>,
27    /// Map from generator index to its current bin index.
28    pub generator_bin_ids: Vec<usize>,
29    /// Precomputed search order for visiting neighboring bins.
30    pub bin_search_order: Vec<(isize, isize, f64)>,
31}
32
33impl AlgorithmGrid2D {
34    /// Creates a new `AlgorithmGrid2D` with the specified dimensions and bounds.
35    pub fn new(nx: usize, ny: usize, bounds: &BoundingBox<2>) -> Self {
36        let sx = (nx as f64) / (bounds.max[0] - bounds.min[0]);
37        let sy = (ny as f64) / (bounds.max[1] - bounds.min[1]);
38
39        let cell_size_x = 1.0 / sx;
40        let cell_size_y = 1.0 / sy;
41
42        let mut bin_search_order = Vec::new();
43        let rx = nx as isize;
44        let ry = ny as isize;
45
46        let get_min_dist_sq = |dx: isize, dy: isize| {
47            let mx = if dx > 0 { (dx - 1) as f64 * cell_size_x } else if dx < 0 { (-dx - 1) as f64 * cell_size_x } else { 0.0 };
48            let my = if dy > 0 { (dy - 1) as f64 * cell_size_y } else if dy < 0 { (-dy - 1) as f64 * cell_size_y } else { 0.0 };
49            mx * mx + my * my
50        };
51
52        for y in -ry..=ry {
53            for x in -rx..=rx {
54                let dist_sq = get_min_dist_sq(x, y);
55                bin_search_order.push((x, y, dist_sq));
56            }
57        }
58        bin_search_order.sort_unstable_by(|a, b| a.2.partial_cmp(&b.2).unwrap_or(std::cmp::Ordering::Equal));
59
60        AlgorithmGrid2D {
61            grid_res_x: nx,
62            grid_res_y: ny,
63            grid_scale_x: sx,
64            grid_scale_y: sy,
65            grid_limit_x: (nx as f64) - 1e-5,
66            grid_limit_y: (ny as f64) - 1e-5,
67            min_x: bounds.min[0],
68            min_y: bounds.min[1],
69            grid_bins: vec![Vec::new(); nx * ny],
70            generator_bin_ids: Vec::new(),
71            bin_search_order,
72        }
73    }
74
75    /// Calculates the linear index of the bin corresponding to the given coordinates.
76    pub fn get_bin_index(&self, x: f64, y: f64, bounds: &BoundingBox<2>) -> usize {
77        let nx = self.grid_res_x;
78
79        let ix = ((x - bounds.min[0]) * self.grid_scale_x).clamp(0.0, self.grid_limit_x) as usize;
80        let iy = ((y - bounds.min[1]) * self.grid_scale_y).clamp(0.0, self.grid_limit_y) as usize;
81
82        ix + iy * nx
83    }
84}
85
86impl SpatialAlgorithm<2> for AlgorithmGrid2D {
87    fn set_generators(&mut self, generators: &[f64], bounds: &BoundingBox<2>) {
88        let total_bins = self.grid_res_x * self.grid_res_y;
89        self.grid_bins.iter_mut().for_each(|bin| bin.clear());
90        if self.grid_bins.len() != total_bins {
91            self.grid_bins = vec![Vec::new(); total_bins];
92        }
93        
94        let count = generators.len() / 2;
95        self.generator_bin_ids = vec![0; count];
96
97        for i in 0..count {
98            let x = generators[i * 2];
99            let y = generators[i * 2 + 1];
100            let bin_idx = self.get_bin_index(x, y, bounds);
101            
102            self.grid_bins[bin_idx].push(i);
103            self.generator_bin_ids[i] = bin_idx;
104        }
105    }
106
107    fn update_generator(&mut self, index: usize, _old_pos: &[f64; 2], new_pos: &[f64; 2], bounds: &BoundingBox<2>) {
108        let new_bin_idx = self.get_bin_index(new_pos[0], new_pos[1], bounds);
109        let old_bin_idx = self.generator_bin_ids[index];
110
111        if new_bin_idx != old_bin_idx {
112            if let Some(pos) = self.grid_bins[old_bin_idx].iter().position(|&id| id == index) {
113                self.grid_bins[old_bin_idx].swap_remove(pos);
114            }
115            self.grid_bins[new_bin_idx].push(index);
116            self.generator_bin_ids[index] = new_bin_idx;
117        }
118    }
119
120    fn visit_neighbors<F>(
121        &self,
122        generators: &[f64],
123        index: usize,
124        pos: [f64; 2],
125        max_dist_sq: &mut f64,
126        mut visitor: F,
127    ) where
128        F: FnMut(usize, [f64; 2], f64) -> f64,
129    {
130        let bin_idx = self.generator_bin_ids[index];
131        let idx_y = bin_idx / self.grid_res_x;
132        let idx_x = bin_idx % self.grid_res_x;
133
134        let cell_size_x = 1.0 / self.grid_scale_x;
135        let cell_size_y = 1.0 / self.grid_scale_y;
136
137        let rel_x = (pos[0] - self.min_x) * self.grid_scale_x - idx_x as f64;
138        let rel_y = (pos[1] - self.min_y) * self.grid_scale_y - idx_y as f64;
139
140        for &(dx, dy, min_d2) in &self.bin_search_order {
141            if min_d2 > 4.0 * *max_dist_sq {
142                break;
143            }
144
145            let bx = idx_x as isize + dx;
146            let by = idx_y as isize + dy;
147
148            if bx >= 0 && bx < self.grid_res_x as isize &&
149               by >= 0 && by < self.grid_res_y as isize {
150                
151                let dx_dist = if dx > 0 { (dx as f64 - rel_x) * cell_size_x } else if dx < 0 { (-(dx + 1) as f64 + rel_x) * cell_size_x } else { 0.0 };
152                let dy_dist = if dy > 0 { (dy as f64 - rel_y) * cell_size_y } else if dy < 0 { (-(dy + 1) as f64 + rel_y) * cell_size_y } else { 0.0 };
153
154                let dx_bin = dx_dist.max(0.0);
155                let dy_bin = dy_dist.max(0.0);
156                
157                if dx_bin * dx_bin + dy_bin * dy_bin <= 4.0 * *max_dist_sq {
158                    let bin_index = (bx as usize) + (by as usize) * self.grid_res_x;
159                    for &j in &self.grid_bins[bin_index] {
160                        if index == j { continue; }
161                        let ox = generators[j * 2];
162                        let oy = generators[j * 2 + 1];
163                        *max_dist_sq = visitor(j, [ox, oy], *max_dist_sq);
164                    }
165                }
166            }
167        }
168    }
169}

175    #[test]
176    fn test_grid_indexing_2d() {
177        let bounds = BoundingBox::new([0.0, 0.0], [10.0, 10.0]);
178        let grid = AlgorithmGrid2D::new(10, 10, &bounds); // 1x1 cells
179
180        let idx = grid.get_bin_index(0.5, 0.5, &bounds);
181        assert_eq!(idx, 0);
182
183        let idx = grid.get_bin_index(1.5, 0.5, &bounds);
184        assert_eq!(idx, 1);
185        
186        let idx = grid.get_bin_index(0.5, 1.5, &bounds);
187        assert_eq!(idx, 10);
188    }
189
190    #[test]
191    fn test_grid_neighbors_2d() {
192        let bounds = BoundingBox::new([0.0, 0.0], [3.0, 3.0]);
193        let mut grid = AlgorithmGrid2D::new(3, 3, &bounds);
194        let generators = vec![0.5, 0.5, 1.5, 0.5];
195        
196        grid.set_generators(&generators, &bounds);
197        
198        let mut neighbors = Vec::new();
199        let mut max_dist_sq = 2.0;
200        
201        grid.visit_neighbors(&generators, 0, [0.5, 0.5], &mut max_dist_sq, |idx, _, d| {
202            neighbors.push(idx);
203            d
204        });
205        assert!(neighbors.contains(&1));
206    }
