99impl WallGeometry<2> for JsWallGeometry2D {
100    fn contains(&self, point: &[f64; 2]) -> bool {
101        if let Ok(func) = Reflect::get(&self.val, &"contains".into()).and_then(|f| f.dyn_into::<Function>()) {
102            let args = Array::of2(&point[0].into(), &point[1].into());
103            if let Ok(res) = func.apply(&self.val, &args) {
104                return res.as_bool().unwrap_or(false);
105            }
106        }
107        false
108    }
109
110    fn cut(&self, generator: &[f64; 2], callback: &mut dyn FnMut([f64; 2], [f64; 2])) {
111        if let Ok(func) = Reflect::get(&self.val, &"cut".into()).and_then(|f| f.dyn_into::<Function>()) {
112            let args = Array::of2(&generator[0].into(), &generator[1].into());
113            if let Ok(res) = func.apply(&self.val, &args) {
114                if res.is_null() || res.is_undefined() { return; }
115                
116                let process_item = |item: &JsValue| -> Option<([f64; 2], [f64; 2])> {
117                    let p = Reflect::get(item, &"point".into()).ok()?;
118                    let n = Reflect::get(item, &"normal".into()).ok()?;
119                    Some((parse_js_point(&p)?, parse_js_point(&n)?))
120                };
121
122                if Array::is_array(&res) {
123                    let arr: Array = res.dyn_into().unwrap();
124                    for i in 0..arr.length() {
125                        if let Some((p, n)) = process_item(&arr.get(i)) {
126                            callback(p, n);
127                        }
128                    }
129                } else {
130                    if let Some((p, n)) = process_item(&res) {
131                        callback(p, n);
132                    }
133                }
134            }
135        }
136    }
137}
