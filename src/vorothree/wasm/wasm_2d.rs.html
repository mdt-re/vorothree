13/// Represents an axis-aligned bounding box in 2D space.
14#[wasm_bindgen]
15#[derive(Clone, Copy, Debug)]
16pub struct BoundingBox2D {
17    pub min_x: f64,
18    pub min_y: f64,
19    pub max_x: f64,
20    pub max_y: f64,
21}
22
23#[wasm_bindgen]
24impl BoundingBox2D {
25    /// Creates a new 2D bounding box.
26    ///
27    /// @param min_x The minimum x coordinate.
28    /// @param min_y The minimum y coordinate.
29    /// @param max_x The maximum x coordinate.
30    /// @param max_y The maximum y coordinate.
31    #[wasm_bindgen(constructor)]
32    pub fn new(min_x: f64, min_y: f64, max_x: f64, max_y: f64) -> BoundingBox2D {
33        BoundingBox2D { min_x, min_y, max_x, max_y }
34    }
35}

40    }
41}

118impl WallGeometry<2> for JsWallGeometry2D {
119    fn contains(&self, point: &[f64; 2]) -> bool {
120        if let Ok(func) = Reflect::get(&self.val, &"contains".into()).and_then(|f| f.dyn_into::<Function>()) {
121            let args = Array::of2(&point[0].into(), &point[1].into());
122            if let Ok(res) = func.apply(&self.val, &args) {
123                return res.as_bool().unwrap_or(false);
124            }
125        }
126        false
127    }
128
129    fn cut(&self, generator: &[f64; 2], callback: &mut dyn FnMut([f64; 2], [f64; 2])) {
130        if let Ok(func) = Reflect::get(&self.val, &"cut".into()).and_then(|f| f.dyn_into::<Function>()) {
131            let args = Array::of2(&generator[0].into(), &generator[1].into());
132            if let Ok(res) = func.apply(&self.val, &args) {
133                if res.is_null() || res.is_undefined() { return; }
134                
135                let process_item = |item: &JsValue| -> Option<([f64; 2], [f64; 2])> {
136                    let p = Reflect::get(item, &"point".into()).ok()?;
137                    let n = Reflect::get(item, &"normal".into()).ok()?;
138                    Some((parse_js_point(&p)?, parse_js_point(&n)?))
139                };
140
141                if Array::is_array(&res) {
142                    let arr: Array = res.dyn_into().unwrap();
143                    for i in 0..arr.length() {
144                        if let Some((p, n)) = process_item(&arr.get(i)) {
145                            callback(p, n);
146                        }
147                    }
148                } else {
149                    if let Some((p, n)) = process_item(&res) {
150                        callback(p, n);
151                    }
152                }
153            }
154        }
155    }
156}

160/// Represents a 2D Voronoi cell.
161#[wasm_bindgen(js_name = Cell2D)]
162pub struct Cell2DWASM {
163    inner: Cell2D,
164}
165
166#[wasm_bindgen(js_class = Cell2D)]
167impl Cell2DWASM {
168    /// The index of the generator corresponding to this cell.
169    #[wasm_bindgen(getter)]
170    pub fn id(&self) -> usize { self.inner.id() }
171    /// The vertices of the cell as a flat array [x0, y0, x1, y1, ...].
172    #[wasm_bindgen(getter)]
173    pub fn vertices(&self) -> Vec<f64> { self.inner.vertices() }
174    /// The indices of the neighboring generators for each edge.
175    #[wasm_bindgen(getter)]
176    pub fn edge_neighbors(&self) -> Vec<i32> { self.inner.edge_neighbors() }
177    /// Calculates the area of the cell.
178    pub fn area(&self) -> f64 { self.inner.area() }
179    /// Calculates the centroid of the cell.
180    pub fn centroid(&self) -> Vec<f64> { self.inner.centroid().to_vec() }
181}

185/// The main 2D Voronoi tessellation class.
186#[wasm_bindgen(js_name = Tessellation2D)]
187pub struct Tessellation2D {
188    inner: Tessellation<2, Cell2D, Algorithm2DGrid>,
189}
190
191#[wasm_bindgen(js_class = Tessellation2D)]
192impl Tessellation2D {
193    /// Creates a new 2D tessellation.
194    ///
195    /// @param bounds The bounding box of the simulation.
196    /// @param nx The number of grid bins in the x direction.
197    /// @param ny The number of grid bins in the y direction.
198    #[wasm_bindgen(constructor)]
199    pub fn new(bounds: BoundingBox2D, nx: usize, ny: usize) -> Tessellation2D {
200        let b: BoundingBox<2> = bounds.into();
201        Tessellation2D { inner: Tessellation::new(b, Algorithm2DGrid::new(nx, ny, &b)) }
202    }
203    /// Sets the generator points.
204    ///
205    /// @param generators A flat array of coordinates [x0, y0, x1, y1, ...].
206    pub fn set_generators(&mut self, generators: &[f64]) { self.inner.set_generators(generators); }
207    /// Updates a specific generator's position.
208    pub fn set_generator(&mut self, index: usize, x: f64, y: f64) { self.inner.set_generator(index, &[x, y]); }
209    /// Generates random points within the bounds and walls.
210    pub fn random_generators(&mut self, count: usize) { self.inner.random_generators(count); }
211    /// Reads generators from a string representation.
212    ///
213    /// Each line should contain an ID followed by coordinates (e.g., "id x y").
214    pub fn read_generators(&mut self, input: &str) { self.inner.read_generators(input); }
215    /// Adds a wall to the tessellation.
216    pub fn add_wall(&mut self, mut wall: Wall2D) { if let Some(w) = wall.take_inner() { self.inner.add_wall(w); } }
217    /// Removes all walls.
218    pub fn clear_walls(&mut self) { self.inner.clear_walls(); }
219    /// Calculates the Voronoi tessellation.
220    pub fn calculate(&mut self) { self.inner.calculate(); }
221    /// Performs one step of Lloyd's relaxation to smooth the cell distribution.
222    pub fn relax(&mut self) { self.inner.relax(); }
223    /// Returns the number of generators.
224    #[wasm_bindgen(getter)]
225    pub fn count_generators(&self) -> usize { self.inner.count_generators() }
226    /// Returns the number of computed cells.
227    #[wasm_bindgen(getter)]
228    pub fn count_cells(&self) -> usize { self.inner.count_cells() }
229    /// Gets a generator's position by index.
230    pub fn get_generator(&self, index: usize) -> Vec<f64> { self.inner.get_generator(index).to_vec() }
231    /// Gets a cell by index.
232    pub fn get_cell(&self, index: usize) -> Option<Cell2DWASM> { self.inner.get_cell(index).map(|inner| Cell2DWASM { inner }) }
233    /// Returns all generators as a flat array.
234    #[wasm_bindgen(getter)]
235    pub fn generators(&self) -> Vec<f64> { self.inner.generators() }
236    /// Returns all cells.
237    #[wasm_bindgen(getter)]
238    pub fn cells(&self) -> Vec<Cell2DWASM> { self.inner.cells().into_iter().map(|inner| Cell2DWASM { inner }).collect() }
