130    pub fn new_bezier(points: &[f64], radius: f64, resolution: usize, closed: bool, id: i32) -> Wall3D {
131        if points.len() != 12 {
132            panic!("Cubic Bezier curve requires exactly 4 control points (12 coordinates)");
133        }
134        let p0 = [points[0], points[1], points[2]];
135        let p1 = [points[3], points[4], points[5]];
136        let p2 = [points[6], points[7], points[8]];
137        let p3 = [points[9], points[10], points[11]];
138        Wall3D { inner: Some(Wall::new(id, Box::new(CubicBezierGeometry::new(p0, p1, p2, p3, radius, resolution, closed)))) }
139    }
140
141    pub fn new_catmull_rom(points: &[f64], radius: f64, resolution: usize, closed: bool, id: i32) -> Wall3D {
142        if points.len() % 3 != 0 {
143            panic!("Catmull-Rom curve points must be a multiple of 3 coordinates");
144        }
145        let mut control_points = Vec::with_capacity(points.len() / 3);
146        for i in (0..points.len()).step_by(3) {
147            control_points.push([points[i], points[i+1], points[i+2]]);
148        }
149        Wall3D { inner: Some(Wall::new(id, Box::new(CatmullRomGeometry::new(control_points, radius, resolution, closed)))) }
150    }

172impl WallGeometry<3> for JsWallGeometry3D {
173    fn contains(&self, point: &[f64; 3]) -> bool {
174        if let Ok(func) = Reflect::get(&self.val, &"contains".into()).and_then(|f| f.dyn_into::<Function>()) {
175            let args = Array::of3(&point[0].into(), &point[1].into(), &point[2].into());
176            if let Ok(res) = func.apply(&self.val, &args) {
177                return res.as_bool().unwrap_or(false);
178            }
179        }
180        false
181    }
182
183    fn cut(&self, generator: &[f64; 3], callback: &mut dyn FnMut([f64; 3], [f64; 3])) {
184        if let Ok(func) = Reflect::get(&self.val, &"cut".into()).and_then(|f| f.dyn_into::<Function>()) {
185            let args = Array::of3(&generator[0].into(), &generator[1].into(), &generator[2].into());
186            if let Ok(res) = func.apply(&self.val, &args) {
187                if res.is_null() || res.is_undefined() { return; }
188                
189                let process_item = |item: &JsValue| -> Option<([f64; 3], [f64; 3])> {
190                    let p = Reflect::get(item, &"point".into()).ok()?;
191                    let n = Reflect::get(item, &"normal".into()).ok()?;
192                    Some((parse_js_point(&p)?, parse_js_point(&n)?))
193                };
194
195                if Array::is_array(&res) {
196                    let arr: Array = res.dyn_into().unwrap();
197                    for i in 0..arr.length() {
198                        if let Some((p, n)) = process_item(&arr.get(i)) {
199                            callback(p, n);
200                        }
201                    }
202                } else {
203                    if let Some((p, n)) = process_item(&res) {
204                        callback(p, n);
205                    }
206                }
207            }
208        }
209    }
210}

233    pub fn face_area(&self, face_index: usize) -> f64 { self.inner.face_area(face_index) }
234    // Workaround for the fact that wasm-bindgen does not support nested vectors directly
235    #[wasm_bindgen(js_name = faces)]
236    pub fn wasm_faces(&self) -> Array {
237        let counts = &self.inner.face_counts;
238        let indices = &self.inner.face_indices;
239        let result = Array::new_with_length(counts.len() as u32);
240        let mut offset = 0;
241        for (i, &count) in counts.iter().enumerate() {
242            let count = count as usize;
243            let end = offset + count;
244            let face_slice = &indices[offset..end];
245            let js_face = Uint16Array::from(face_slice);
246            result.set(i as u32, js_face.into());
247            offset = end;
248        }
249        result
250    }
251}
