11/// The starting ID for walls. Wall IDs must be less than or equal to this value
12/// to avoid conflicts with non-negative generator IDs and the bounding box IDs.
13pub const WALL_ID_START: i32 = -10;
14
15/// Trait defining the geometry and logic of a wall.
16/// Must be Send + Sync to support parallel execution in Tessellation.
17pub trait WallGeometry: Send + Sync + std::fmt::Debug {
18    /// Checks if a point is inside the valid region defined by the wall.
19    fn contains(&self, point: &[f64; 3]) -> bool;
20
21    /// Calculates the clipping plane for a given generator.
22    /// Returns a tuple (point_on_plane, plane_normal).
23    /// The normal should point OUT of the valid region (towards the region to be clipped).
24    fn cut(&self, generator: &[f64; 3], callback: &mut dyn FnMut([f64; 3], [f64; 3]));
25}
26
27/// A clipping boundary for the Voronoi tessellation.
28///
29/// A `Wall` is a container for a `WallGeometry` implementation, giving it a unique
30/// integer ID. This ID will be reported in the `face_neighbors` array of a `Cell`
31/// for faces that have been clipped by this wall.
32#[wasm_bindgen]
33pub struct Wall {
34    id: i32,
35    inner: Box<dyn WallGeometry>,
36}
37
38impl Wall {
39    /// Creates a new `Wall` from a Rust struct that implements the `WallGeometry` trait.
40    pub fn new(id: i32, geometry: Box<dyn WallGeometry>) -> Wall {
41        if id > WALL_ID_START {
42            panic!("Wall ID must be <= {}", WALL_ID_START);
43        }
44        Wall {
45            id,
46            inner: geometry,
47        }
48    }
49}

134    pub fn new_bezier(points: &[f64], radius: f64, resolution: usize, closed: bool, id: i32) -> Wall {
135        if points.len() != 12 {
136            panic!("Cubic Bezier curve requires exactly 4 control points (12 coordinates)");
137        }
138        let p0 = [points[0], points[1], points[2]];
139        let p1 = [points[3], points[4], points[5]];
140        let p2 = [points[6], points[7], points[8]];
141        let p3 = [points[9], points[10], points[11]];
142        Wall::new(id, Box::new(CubicBezierGeometry::new(p0, p1, p2, p3, radius, resolution, closed)))
143    }

145    pub fn new_catmull_rom(points: &[f64], radius: f64, resolution: usize, closed: bool, id: i32) -> Wall {
146        if points.len() % 3 != 0 {
147            panic!("Catmull-Rom curve points must be a multiple of 3 coordinates");
148        }
149        let mut control_points = Vec::with_capacity(points.len() / 3);
150        for i in (0..points.len()).step_by(3) {
151            control_points.push([points[i], points[i+1], points[i+2]]);
152        }
153        Wall::new(id, Box::new(CatmullRomGeometry::new(control_points, radius, resolution, closed)))
154    }

165/// Wrapper for JS-defined walls.
166/// SAFETY: This claims to be Send+Sync to satisfy Rayon, but calling JS from workers is dangerous.
167/// Users should ensure they don't use parallel execution if using JS walls, or that the runtime supports it.
168struct JsWallGeometry {
169    val: JsValue,
170}
171
172unsafe impl Send for JsWallGeometry {}
173unsafe impl Sync for JsWallGeometry {}
174
175impl std::fmt::Debug for JsWallGeometry {
176    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
177        write!(f, "JsWallGeometry")
178    }
179}
180
181impl WallGeometry for JsWallGeometry {
182    fn contains(&self, point: &[f64; 3]) -> bool {
183        if let Ok(func) = Reflect::get(&self.val, &"contains".into()).and_then(|f| f.dyn_into::<Function>()) {
184            let args = Array::of3(&point[0].into(), &point[1].into(), &point[2].into());
185            if let Ok(res) = func.apply(&self.val, &args) {
186                return res.as_bool().unwrap_or(false);
187            }
188        }
189        false
190    }
191
192    fn cut(&self, generator: &[f64; 3], callback: &mut dyn FnMut([f64; 3], [f64; 3])) {
193        if let Ok(func) = Reflect::get(&self.val, &"cut".into()).and_then(|f| f.dyn_into::<Function>()) {
194            let args = Array::of3(&generator[0].into(), &generator[1].into(), &generator[2].into());
195            if let Ok(res) = func.apply(&self.val, &args) {
196                if res.is_null() || res.is_undefined() { return; }
197                
198                if Array::is_array(&res) {
199                    let arr: Array = res.dyn_into().unwrap();
200                    for i in 0..arr.length() {
201                        let item = arr.get(i);
202                        if let Some((p, n)) = parse_js_cut_result(&item) {
203                            callback(p, n);
204                        }
205                    }
206                } else {
207                    if let Some((p, n)) = parse_js_cut_result(&res) {
208                        callback(p, n);
209                    }
210                }
211            }
212        }
213    }
214}

216/// Parses the result from a JavaScript `cut` function call.
217fn parse_js_cut_result(val: &JsValue) -> Option<([f64; 3], [f64; 3])> {
218    let p = Reflect::get(val, &"point".into()).ok().and_then(|v| v.dyn_into::<Array>().ok())?;
219    let n = Reflect::get(val, &"normal".into()).ok().and_then(|v| v.dyn_into::<Array>().ok())?;
220    Some(([p.get(0).as_f64()?, p.get(1).as_f64()?, p.get(2).as_f64()?], [n.get(0).as_f64()?, n.get(1).as_f64()?, n.get(2).as_f64()?]))
221}
